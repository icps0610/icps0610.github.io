```	ruby
#http://standards-oui.ieee.org/oui.txt

m = (0..15).map{|i|i.to_s(16).upcase}
random_mac = (0..2).map{":"+(0..1).map{m.sample}.join}.join

co = []
File.readlines('oui.txt').each do |i|
  mac = i.split("\n").join.scan(/(\w{2}-\w{2}-\w{2})   \(hex\)(.*)/)[0]
  co <<  [mac[0].gsub('-', ':'), mac[1].strip] if mac
end

mac = co.sample
puts mac[1]
puts mac[0]+random_mac
```

``` python
import time
import datetime

u = "1284101485"
s = datetime.datetime.fromtimestamp( int(u)).strftime('%Y-%m-%d %H:%M:%S')
print(s)

u = time.mktime(datetime.datetime.strptime(s, '%Y-%m-%d %H:%M:%S').timetuple())
print(u)
```

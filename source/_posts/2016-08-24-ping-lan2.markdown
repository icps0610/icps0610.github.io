``` bat
@echo off
SETLOCAL ENABLEDELAYEDEXPANSION
set lan=192.168.0
set start=%1
set stop=%2
for /l %%i in (%start%,1,%stop%) do (
   ping -n 1 -w 1 %lan%.%%i > nul
    if !errorlevel! == 0 (
        echo %lan%.%%i O
    ) else (
        echo %lan%.%%i X
    )
)
```

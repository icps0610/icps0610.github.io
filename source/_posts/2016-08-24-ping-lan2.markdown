``` bat
@echo off
SETLOCAL ENABLEDELAYEDEXPANSION
for /l %%i in (1,1,254) do (
   ping -n 1 -w 1 192.168.0.%%i > nul & if !errorlevel! == 0 echo 192.168.0.%%i
)
```

``` bat
@echo off
SETLOCAL ENABLEDELAYEDEXPANSION
```

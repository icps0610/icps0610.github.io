``` ruby
require 'rss'
require 'open-uri'

url = 'http://www.appledaily.com.tw/rss/newcreate/kind/rnews/type/new'
open(url) do |rss|
  feed = RSS::Parser.parse(rss)
  feed.items.each do |item|
    puts "Title: #{item.title}"
  end
end

```

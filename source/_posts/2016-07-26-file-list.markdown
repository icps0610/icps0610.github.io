def dir(file)
  file.each do |f|
    if File.directory?(f) == true
      file=`ls #{f}`.split("\n").map{|i|f+"\\"+i}
      dir(file)
    else
      `ls #{f}`.split("\n").each do |i|
        system "rm #{i}"
        system "touch #{i}"
      end
    end
  end
end

``` python
import sys, getopt
#args = " " . join(map(lambda s: '"%s"' % s if ' ' in s else s, sys.argv[2:]))

opts, args = getopt.getopt(sys.argv[1:], "hm:r:")
mode = ""
rating = ""
for op, value in opts:
  if op == "-m":
    mode = value
  elif op == "-r":
    rating = value
  else:
  	print("-m mode -r rating")
  	sys.exit()
print(mode)
print(rating)
```

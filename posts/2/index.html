把wifi網卡當作基地台

@echo off
netsh wlan set hostednetwork mode=allow ssid=wifi-name key=12345
netsh wlan start hostednetwork
echo ==============已經啟動 按任意鍵取消==============
pause>nul
netsh wlan stop hostednetwork

顯示wifi密碼

netsh wlan show profile name=wifi-name key=clear
